#### What this PR does / why we need it:
The PR introduces MVP version of ConcurrentAdmission feature. It includes
- new API definition
- new Concurrent Admission Controller
- minor changes to scheduling logic to narrow choice of resource flavor that are allowed to be assigned to a particular Workload
- minor changes to cache
- validation webhooks

The PR introduces two sub-types of Workloads as described in #8861 : 1) Parents; 2) Variants

The high-level idea is that the new concurrent admission controller creates Variants which are "copies" of a Parent Workload but with constraints regarding scheduling on a particular flavor.

The Variants are processed by Kueue almost as if they were regular Workloads, with constraints:
- Kueue scheduler can only assign certain resource flavor to them;
- Jobframework reconciler is not aware of them, and don't process them at all - this is to keep the implementation simple, and keep the current invariant, that a job corresponds to a Workload in 1;1 manner.

Variants are keepers of quota reservation and admission checks. 

The Parents are vastly different than regular Workloads:
- They do not undergo scheduling cycle
- They are not part of the Kueue queue cache (with a small exception) and scheduling cache
- They serve as an aggregation point between the Job and its Variants

Information between Parents and Variants is synced by the variant controller, and it's done both directions: a Parent to Variants, and Variants to Parent

Synchronization from Variants to Parent:
- If a Variant gets admitted, it sets proper conditions on its Parent as well
- If a Variant gets evicted, it sets proper conditions on its Parent as well

Synchronization from Parent to Variants:
- If a Parent gets deactivated, it deactivates its Variants
- If a Parent is finished, it finishes its Variants
- If a Parent has WaitForPodsReady condition, it syncs it on its Variants

This PR is posted as a draft for now as I'm working on integration and e2e tests - they might be added in a follow-up not to blow the size of a single PR. I've tested this code e2e manually as well.

A few follow-ups this MVP requires (I'm going to create GH issues for them):
- Create index for Variants
- Include the scheduling constraints in the hashing mechanism
- Add logic to scheduler to preempt less favorable siblings before admission the more favorable one (currently concurrent admission controller deactivates less favorable sibling)
- Add logic to scheduler to ensure only one Variant per siblings can preempt other Workloads to prevent excessive preemption 
- Make CQ's config immutable (add listener on CQs in concurrent admission controller)
- Aggregate `SchedulingStats` in Parent
- Sync `requeueAt` field
- Sync Variants with changes to CQ's ResourceGroups - in particular delete irrelevant ones

+	if parent == nil {

+	}

+	flavorOrder := make(map[kueue.ResourceFlavorReference]int, len(cq.Spec.ResourceGroups[0].Flavors))

+		flavorOrder[flavor.Name] = i

+	log = log.WithValues("clusterQueue", cq.Name)

+		if err := r.createVariants(ctx, parent, variants, flavorOrder); err != nil {
+			return ctrl.Result{}, err
+		}
+	}
+	if len(variants) == len(flavorOrder) {
+		log.V(3).Info("Desired number of variants, no action needed", "desired", len(flavorOrder), "actual", len(variants))
+	}
+
+	log.V(3).Info("Syncing variants if needed")
+	if err := r.syncVariantEvictionStatus(ctx, parent, variants); err != nil {
+		log.V(3).Info("Failed to sync variant eviction status", "error", err)

+	}

+	log.V(3).Info("Deactivating variants if needed")

+		log.V(3).Info("Failed to deactivate variants", "error", err)
+		return ctrl.Result{}, err
+	}
+
+	log.V(3).Info("Activating variants if needed")
+	if err := r.activateVariants(ctx, parent, variants, cq, flavorOrder); err != nil {
+		log.V(3).Info("Failed to activate variants", "error", err)

+	}

+	if err := r.syncAdmissionStatus(ctx, parent, variants); err != nil {

+func sortVariantsByFlavorOrder(variants []kueue.Workload, flavorOrder map[kueue.ResourceFlavorReference]int) []kueue.Workload {
+	slices.SortFunc(variants, func(a, b kueue.Workload) int {

+		bFlavor := getVariantFlavor(&b)

+func (r *variantReconciler) syncVariantEvictionStatus(ctx context.Context, parent *kueue.Workload, variants []kueue.Workload) error {
+	log := ctrl.LoggerFrom(ctx)
+	log.V(3).Info("Syncing eviction status of variants", "parent", parent.Name)
+	for i := range variants {
+		v := &variants[i]
+		evCond := apimeta.FindStatusCondition(v.Status.Conditions, kueue.WorkloadEvicted)
+		if evCond != nil && evCond.Status == metav1.ConditionTrue && workload.HasQuotaReservation(v) {
+			log.V(2).Info("The variant has been evicted, clearing parents's admission", "variant", v.Name, "parent", parent.Name)
+			if err := r.clearWorkloadAdmission(ctx, v, evCond); err != nil {
+				return fmt.Errorf("clearing variant admission: %w", err)
+			}
+		}
+	}
+	return nil

+func (r *variantReconciler) clearWorkloadAdmission(ctx context.Context, wl *kueue.Workload, evCond *metav1.Condition) error {
+	return workload.PatchAdmissionStatus(ctx, r.client, wl, r.clock, func(w *kueue.Workload) (bool, error) {
+		setRequeued := (evCond.Reason == kueue.WorkloadEvictedByPreemption) ||
+			(evCond.Reason == kueue.WorkloadEvictedDueToNodeFailures)
+		updated := workload.SetRequeuedCondition(w, evCond.Reason, evCond.Message, setRequeued)
+		if workload.UnsetQuotaReservationWithCondition(w, "Pending", evCond.Message, r.clock.Now()) {
+			updated = true
+		}
+		return updated, nil
+	})
+}

+func (r *variantReconciler) deactivateVariant(ctx context.Context, v *kueue.Workload) error {
+	log := ctrl.LoggerFrom(ctx)
+	if err := workload.Deactivate(ctx, r.client, v); err != nil {
+		return err
+	}
+	// fetch the updated variant and unset quota
+	if err := r.client.Get(ctx, client.ObjectKeyFromObject(v), v); err != nil {
+		return err
+	}
+	if evCond := apimeta.FindStatusCondition(v.Status.Conditions, kueue.WorkloadEvicted); evCond != nil && evCond.Status == metav1.ConditionTrue {
+		if workload.HasQuotaReservation(v) {
+			log.V(3).Info("The variant is no longer active, clear the workloads admission")
+			err := workload.PatchAdmissionStatus(ctx, r.client, v, r.clock, func(wl *kueue.Workload) (bool, error) {
+				// The requeued condition status set to true only on EvictedByPreemption
+				setRequeued := (evCond.Reason == kueue.WorkloadEvictedByPreemption) || (evCond.Reason == kueue.WorkloadEvictedDueToNodeFailures)
+				updated := workload.SetRequeuedCondition(wl, evCond.Reason, evCond.Message, setRequeued)
+				if workload.UnsetQuotaReservationWithCondition(wl, "Pending", evCond.Message, r.clock.Now()) {
+					updated = true
+				}
+				return updated, nil
+			})
+			if err != nil {
+				return fmt.Errorf("clearing admission: %w", err)
+			}
+		}
+	}
+	return nil

+func (r *variantReconciler) deactivateVariants(
+	ctx context.Context,
+	parent *kueue.Workload,
+	variants []kueue.Workload,
+	cq *kueue.ClusterQueue,
+	flavorOrder map[kueue.ResourceFlavorReference]int,
+) error {
+	log := ctrl.LoggerFrom(ctx)
+	if !workload.IsActive(parent) {
+		log.V(2).Info("Parent is not active, deactivating all variants", "parent", parent.Name)
+		for i := range variants {
+			v := &variants[i]
+			if err := r.deactivateVariant(ctx, v); err != nil {
+				return err
+			}
+		}
+		return nil
+	}
+
+	admittedWl := getAdmittedVariant(variants)
+	if admittedWl == nil {
+		log.V(3).Info("No admitted variant, no need to deactivate any variant")
+		return nil
+	}
+	// deactivate Variants below minTargetFlavor if specified
+	var minTargetFlavor *kueue.ResourceFlavorReference
+	if cq.Spec.ConcurrentAdmissionPolicy != nil && cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints != nil {
+		minTargetFlavor = cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints.MinPreferredFlavorName
+	}
+	if minTargetFlavor != nil {
+		log.V(3).Info("Deactivating variants below minTargetFlavor", "minTargetFlavor", *minTargetFlavor)
+		for i := range variants {
+			v := &variants[i]
+			if v.Name == admittedWl.Name {
+				continue
+			}
+			if flavorOrder[getVariantFlavor(v)] > flavorOrder[*minTargetFlavor] {
+				log.V(2).Info("Deactivating variant because it is below the minTargetFlavor", "variant", v.Name, "flavor", getVariantFlavor(v), "minTargetFlavor", *minTargetFlavor)
+				if err := r.deactivateVariant(ctx, v); err != nil {
+					return err
+				}
+			}
+		}
+	}
+	// also deactivate Variants below the admitted variant regardless of minTargetFlavor
+	log.V(3).Info("Deactivating variants below the admitted variant", "admittedVariant", admittedWl.Name, "admittedFlavor", getVariantFlavor(admittedWl))
+	for i := range variants {
+		v := &variants[i]
+		if flavorOrder[getVariantFlavor(v)] > flavorOrder[getVariantFlavor(admittedWl)] {
+			log.V(2).Info("Deactivating variant because it is below the admitted variant", "variant", v.Name, "flavor", getVariantFlavor(v), "admittedFlavor", getVariantFlavor(admittedWl))
+			if err := r.deactivateVariant(ctx, v); err != nil {
+				return err
+			}
+		}
+	}
+	return nil
+}

+func (r *variantReconciler) activateVariants(ctx context.Context, parent *kueue.Workload, variants []kueue.Workload, cq *kueue.ClusterQueue, flavorOrder map[kueue.ResourceFlavorReference]int) error {
+	log := ctrl.LoggerFrom(ctx)
+	if !workload.IsActive(parent) {
+		log.V(3).Info("Parent is not active, no needed to activate variants", "parent", parent.Name)
+		return nil
+	}
+	admittedVariant := getAdmittedVariant(variants)
+	if admittedVariant == nil {
+		log.V(3).Info("No admitted variant, activating all variants")
+		// no admitted variants so activate all variants if they are not active, case of preemption
+		for i := range variants {
+			v := &variants[i]
+			if err := workload.Activate(ctx, r.client, v); err != nil {
+				return err
+			}
+		}
+		return nil
+	}

+	var minTargetFlavor *kueue.ResourceFlavorReference
+	if cq.Spec.ConcurrentAdmissionPolicy != nil && cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints != nil {
+		minTargetFlavor = cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints.MinPreferredFlavorName
+	}
+	if minTargetFlavor != nil {

+			v := &variants[i]

+		v := &variants[i]
+		if flavorOrder[getVariantFlavor(v)] < flavorOrder[getVariantFlavor(admittedVariant)] {

+func (r *variantReconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
+	log := ctrl.LoggerFrom(ctx)
+	log.V(2).Info("Reconcile Workload")
+
+	parent, variants, cq, err := r.getFamilyAndClusterQueue(ctx, req)
+	if err != nil {
+		return ctrl.Result{}, err
+	}
+	if parent == nil {
+		return ctrl.Result{}, nil
+	}
+
+	flavorOrder := make(map[kueue.ResourceFlavorReference]int, len(cq.Spec.ResourceGroups[0].Flavors))
+	for i, flavor := range cq.Spec.ResourceGroups[0].Flavors {
+		flavorOrder[flavor.Name] = i
+	}
+	variants = sortVariantsByFlavorOrder(variants, flavorOrder)
+
+	log = log.WithValues("clusterQueue", cq.Name)
+	log.V(2).Info("Found Workload family and ClusterQueue", "variants", klog.KObjSlice(variants))
+
+	// TODO: If ConcurrentAdmission is no longer enabled for this CQ, delete parent and variants.
+
+	if len(variants) < len(flavorOrder) {
+		log.V(3).Info("Too few variants, creating new ones", "desired", len(flavorOrder), "actual", len(variants))
+		if err := r.createVariants(ctx, parent, variants, flavorOrder); err != nil {
+			return ctrl.Result{}, err
+		}
+	}
+	if len(variants) == len(flavorOrder) {
+		log.V(3).Info("Desired number of variants, no action needed", "desired", len(flavorOrder), "actual", len(variants))
+	}
+
+	log.V(3).Info("Syncing variants if needed")
+	if err := r.syncVariantEvictionStatus(ctx, parent, variants); err != nil {
+		log.V(3).Info("Failed to sync variant eviction status", "error", err)
+		return ctrl.Result{}, err
+	}
+
+	log.V(3).Info("Deactivating variants if needed")
+	if err := r.deactivateVariants(ctx, parent, variants, cq, flavorOrder); err != nil {
+		log.V(3).Info("Failed to deactivate variants", "error", err)
+		return ctrl.Result{}, err
+	}
+
+	log.V(3).Info("Activating variants if needed")
+	if err := r.activateVariants(ctx, parent, variants, cq, flavorOrder); err != nil {
+		log.V(3).Info("Failed to activate variants", "error", err)
+		return ctrl.Result{}, err
+	}
+
+	if err := r.syncAdmissionStatus(ctx, parent, variants); err != nil {
+		log.V(3).Info("Failed to sync admission status", "error", err)
+		return ctrl.Result{}, err
+	}
+	return ctrl.Result{}, nil

+	cq, err := r.getClusterQueue(ctx, wl)

+		return nil, nil, nil, err

+func (r *variantReconciler) getVariantsForParent(ctx context.Context, parent *kueue.Workload) ([]kueue.Workload, error) {

+	list := &kueue.WorkloadList{}

+		// TODO: Index variants
+		return nil, err

+	for i := range list.Items {
+		if getParentWorkload(&list.Items[i]) == parent.Name {
+			variants = append(variants, list.Items[i])

+		Status: parent.Status,

+func sortVariantsByFlavorOrder(variants []kueue.Workload, flavorOrder map[kueue.ResourceFlavorReference]int) []kueue.Workload {
+	slices.SortFunc(variants, func(a, b kueue.Workload) int {
+		aFlavor := getVariantFlavor(&a) // we only support one flavor per variant for now
+		bFlavor := getVariantFlavor(&b)
+		return flavorOrder[aFlavor] - flavorOrder[bFlavor]
+	})
+	return variants

+func (r *variantReconciler) syncVariantEvictionStatus(ctx context.Context, parent *kueue.Workload, variants []kueue.Workload) error {
+	log := ctrl.LoggerFrom(ctx)
+	log.V(3).Info("Syncing eviction status of variants", "parent", parent.Name)
+	for i := range variants {
+		v := &variants[i]
+		evCond := apimeta.FindStatusCondition(v.Status.Conditions, kueue.WorkloadEvicted)
+		if evCond != nil && evCond.Status == metav1.ConditionTrue && workload.HasQuotaReservation(v) {
+			log.V(2).Info("The variant has been evicted, clearing parents's admission", "variant", v.Name, "parent", parent.Name)
+			if err := r.clearWorkloadAdmission(ctx, v, evCond); err != nil {
+				return fmt.Errorf("clearing variant admission: %w", err)
+			}
+		}
+	}
+	return nil

+func (r *variantReconciler) clearWorkloadAdmission(ctx context.Context, wl *kueue.Workload, evCond *metav1.Condition) error {
+	return workload.PatchAdmissionStatus(ctx, r.client, wl, r.clock, func(w *kueue.Workload) (bool, error) {
+		setRequeued := (evCond.Reason == kueue.WorkloadEvictedByPreemption) ||
+			(evCond.Reason == kueue.WorkloadEvictedDueToNodeFailures)
+		updated := workload.SetRequeuedCondition(w, evCond.Reason, evCond.Message, setRequeued)
+		if workload.UnsetQuotaReservationWithCondition(w, "Pending", evCond.Message, r.clock.Now()) {
+			updated = true
+		}
+		return updated, nil
+	})
+}

+func (r *variantReconciler) deactivateVariant(ctx context.Context, v *kueue.Workload) error {
+	log := ctrl.LoggerFrom(ctx)
+	if err := workload.Deactivate(ctx, r.client, v); err != nil {
+		return err
+	}
+	// fetch the updated variant and unset quota
+	if err := r.client.Get(ctx, client.ObjectKeyFromObject(v), v); err != nil {
+		return err
+	}
+	if evCond := apimeta.FindStatusCondition(v.Status.Conditions, kueue.WorkloadEvicted); evCond != nil && evCond.Status == metav1.ConditionTrue {
+		if workload.HasQuotaReservation(v) {
+			log.V(3).Info("The variant is no longer active, clear the workloads admission")
+			err := workload.PatchAdmissionStatus(ctx, r.client, v, r.clock, func(wl *kueue.Workload) (bool, error) {
+				// The requeued condition status set to true only on EvictedByPreemption
+				setRequeued := (evCond.Reason == kueue.WorkloadEvictedByPreemption) || (evCond.Reason == kueue.WorkloadEvictedDueToNodeFailures)
+				updated := workload.SetRequeuedCondition(wl, evCond.Reason, evCond.Message, setRequeued)
+				if workload.UnsetQuotaReservationWithCondition(wl, "Pending", evCond.Message, r.clock.Now()) {
+					updated = true
+				}
+				return updated, nil
+			})
+			if err != nil {
+				return fmt.Errorf("clearing admission: %w", err)
+			}
+		}
+	}
+	return nil

+func (r *variantReconciler) deactivateVariants(
+	ctx context.Context,
+	parent *kueue.Workload,
+	variants []kueue.Workload,
+	cq *kueue.ClusterQueue,
+	flavorOrder map[kueue.ResourceFlavorReference]int,
+) error {
+	log := ctrl.LoggerFrom(ctx)
+	if !workload.IsActive(parent) {
+		log.V(2).Info("Parent is not active, deactivating all variants", "parent", parent.Name)
+		for i := range variants {
+			v := &variants[i]
+			if err := r.deactivateVariant(ctx, v); err != nil {
+				return err
+			}
+		}
+		return nil
+	}
+
+	admittedWl := getAdmittedVariant(variants)
+	if admittedWl == nil {
+		log.V(3).Info("No admitted variant, no need to deactivate any variant")
+		return nil
+	}
+	// deactivate Variants below minTargetFlavor if specified
+	var minTargetFlavor *kueue.ResourceFlavorReference
+	if cq.Spec.ConcurrentAdmissionPolicy != nil && cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints != nil {
+		minTargetFlavor = cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints.MinPreferredFlavorName
+	}
+	if minTargetFlavor != nil {
+		log.V(3).Info("Deactivating variants below minTargetFlavor", "minTargetFlavor", *minTargetFlavor)
+		for i := range variants {
+			v := &variants[i]
+			if v.Name == admittedWl.Name {
+				continue
+			}
+			if flavorOrder[getVariantFlavor(v)] > flavorOrder[*minTargetFlavor] {
+				log.V(2).Info("Deactivating variant because it is below the minTargetFlavor", "variant", v.Name, "flavor", getVariantFlavor(v), "minTargetFlavor", *minTargetFlavor)
+				if err := r.deactivateVariant(ctx, v); err != nil {
+					return err
+				}
+			}
+		}
+	}
+	// also deactivate Variants below the admitted variant regardless of minTargetFlavor
+	log.V(3).Info("Deactivating variants below the admitted variant", "admittedVariant", admittedWl.Name, "admittedFlavor", getVariantFlavor(admittedWl))
+	for i := range variants {
+		v := &variants[i]
+		if flavorOrder[getVariantFlavor(v)] > flavorOrder[getVariantFlavor(admittedWl)] {
+			log.V(2).Info("Deactivating variant because it is below the admitted variant", "variant", v.Name, "flavor", getVariantFlavor(v), "admittedFlavor", getVariantFlavor(admittedWl))
+			if err := r.deactivateVariant(ctx, v); err != nil {
+				return err
+			}
+		}
+	}
+	return nil
+}

+func (r *variantReconciler) activateVariants(ctx context.Context, parent *kueue.Workload, variants []kueue.Workload, cq *kueue.ClusterQueue, flavorOrder map[kueue.ResourceFlavorReference]int) error {
+	log := ctrl.LoggerFrom(ctx)
+	if !workload.IsActive(parent) {
+		log.V(3).Info("Parent is not active, no needed to activate variants", "parent", parent.Name)
+		return nil
+	}
+	admittedVariant := getAdmittedVariant(variants)
+	if admittedVariant == nil {
+		log.V(3).Info("No admitted variant, activating all variants")
+		// no admitted variants so activate all variants if they are not active, case of preemption
+		for i := range variants {
+			v := &variants[i]
+			if err := workload.Activate(ctx, r.client, v); err != nil {
+				return err
+			}
+		}
+		return nil
+	}
+	// activate all variants that are at least at the minTargetFlavor if specificed
+	var minTargetFlavor *kueue.ResourceFlavorReference
+	if cq.Spec.ConcurrentAdmissionPolicy != nil && cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints != nil {
+		minTargetFlavor = cq.Spec.ConcurrentAdmissionPolicy.Migration.Constraints.MinPreferredFlavorName
+	}
+	if minTargetFlavor != nil {
+		for i := range variants {
+			v := &variants[i]
+			if flavorOrder[getVariantFlavor(v)] <= flavorOrder[*minTargetFlavor] && flavorOrder[getVariantFlavor(v)] < flavorOrder[getVariantFlavor(admittedVariant)] {
+				// activate the variant, the smaller or equal the flavor order is to the minTargetFlavor, the higher the priority is
+				if err := workload.Activate(ctx, r.client, v); err != nil {
+					return err
+				}
+			}
+		}
+		return nil
+	}
+	// no minTargetFlavor specified, so activate all variants that are below the admitted variant in the flavor order
+	for i := range variants {
+		v := &variants[i]
+		if flavorOrder[getVariantFlavor(v)] < flavorOrder[getVariantFlavor(admittedVariant)] {
+			// activate the variant, the smaller the flavor order is to the admitted variant, the higher the priority is
+			if err := workload.Activate(ctx, r.client, v); err != nil {
+				return err
+			}
+		}
+	}
+	return nil
+}
