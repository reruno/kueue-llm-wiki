+func (o *WorkloadOptions) parseForObject() error {
+	if o.UserSpecifiedForObject == "" {
+		return nil
+	}
+
+	parts := strings.Split(o.UserSpecifiedForObject, "/")
+	if len(parts) > 2 {
+		return fmt.Errorf("invalid value '%s' used in --for flag; value must be in the format [TYPE[.API-GROUP]/]NAME", o.UserSpecifiedForObject)
+	}
+
+	if len(parts) == 1 {
+		o.ForObject.Name = parts[0]
+		return nil
+	}
+
+	o.ForObject.Name = parts[1]
+	o.ForObject.Type, o.ForObject.APIGroup, _ = strings.Cut(parts[0], ".")
+
+	return nil

+func (o *WorkloadOptions) filterList(list *v1beta1.WorkloadList) {
+	if len(list.Items) == 0 {
+		return
+	}
+	filteredItems := make([]v1beta1.Workload, 0, len(o.LocalQueueFilter))
+	for _, wl := range list.Items {
+		if o.filterByLocalQueue(&wl) && o.filterByClusterQueue(&wl) && o.filterByOnlyPending(&wl) && o.filterByOnlyAdmitted(&wl) &&
+			o.filterByObjectRef(&wl, o.ForObject.Name, isNameEqual) &&
+			o.filterByObjectRef(&wl, o.ForObject.Type, isTypeEqual) &&
+			o.filterByObjectRef(&wl, o.ForObject.APIGroup, isAPIGroupEqual) {
+			filteredItems = append(filteredItems, wl)
+		}
+	}
+	list.Items = filteredItems
