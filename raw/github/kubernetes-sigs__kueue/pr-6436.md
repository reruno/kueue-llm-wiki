+func calculateExpectedPenalty(requests []corev1.ResourceList, afsConfig *config.AdmissionFairSharing) corev1.ResourceList {
+	alpha := admissionfairsharing.CalculateAlphaRate(
+		afsConfig.UsageSamplingInterval.Seconds(),
+		afsConfig.UsageHalfLifeTime.Seconds(),
+	)
+
+	result := corev1.ResourceList{}
+	for _, request := range requests {
+		penalty := utilresource.MulByFloat(request, alpha)
+		result = utilresource.MergeResourceListKeepSum(result, penalty)
+	}
+	return result
