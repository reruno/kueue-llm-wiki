+func TestClusterQueueConversion_RoundTrip(t *testing.T) {
+	testCases := map[string]struct {
+		v1beta1Obj *ClusterQueue
+	}{
+		"complete ClusterQueue with cohort and FlavorFungibility": {
+			v1beta1Obj: &ClusterQueue{

+				Spec: ClusterQueueSpec{

+							Status: metav1.ConditionTrue,
+							Reason: "Ready",
+						},
+					},
+					PendingWorkloads: 10,

+			},
+		},

+			v1beta1Obj: &ClusterQueue{
+				ObjectMeta: metav1.ObjectMeta{
+					Name: "cq-with-checks",

+				Spec: ClusterQueueSpec{
+					AdmissionChecks: []AdmissionCheckReference{"check1", "check2"},
+				},
+			},
+		},
+		"ClusterQueue with deprecated Preempt and Borrow": {
+			v1beta1Obj: &ClusterQueue{
+				ObjectMeta: metav1.ObjectMeta{
+					Name: "deprecated-cq",
+				},
+				Spec: ClusterQueueSpec{
+					FlavorFungibility: &FlavorFungibility{
+						WhenCanPreempt: Preempt,
+						WhenCanBorrow:  Borrow,
+					},
+				},
+			},
+		},
+		"minimal ClusterQueue": {
+			v1beta1Obj: &ClusterQueue{
+				ObjectMeta: metav1.ObjectMeta{
+					Name: "minimal-cq",
+				},
+			},
+		},
+	}

+	for name, tc := range testCases {
+		t.Run(name, func(t *testing.T) {
+			// Convert v1beta1 -> v1beta2
+			v1beta2Obj := &v1beta2.ClusterQueue{}
+			if err := tc.v1beta1Obj.ConvertTo(v1beta2Obj); err != nil {
+				t.Fatalf("ConvertTo failed: %v", err)
+			}
+
+			// Convert v1beta2 -> v1beta1 (round-trip)
+			roundTripped := &ClusterQueue{}
+			if err := roundTripped.ConvertFrom(v1beta2Obj); err != nil {
+				t.Fatalf("ConvertFrom failed: %v", err)
+			}
+
+			// Verify round-trip
+			// Note: For deprecated values (Preempt, Borrow), they convert to MayStopSearch,
+			// so round-trip will show MayStopSearch instead of the original deprecated values
+			switch name {
+			case "ClusterQueue with deprecated Preempt and Borrow":
+				// Special handling for deprecated values
+				expected := tc.v1beta1Obj.DeepCopy()
+				expected.Spec.FlavorFungibility.WhenCanPreempt = MayStopSearch
+				expected.Spec.FlavorFungibility.WhenCanBorrow = MayStopSearch
+				if diff := cmp.Diff(expected, roundTripped); diff != "" {
+					t.Errorf("round-trip conversion produced diff (-expected +roundtripped):\n%s", diff)
+				}
+			case "ClusterQueue with AdmissionChecks":
+				// AdmissionChecks convert to AdmissionChecksStrategy in v1beta2.
+				// On the way back, AdmissionChecks are lost but AdmissionChecksStrategy is populated.
+				expected := tc.v1beta1Obj.DeepCopy()
+				expected.Spec.AdmissionChecks = nil
+				// Also need to copy the AdmissionChecksStrategy that gets populated on round-trip
+				expected.Spec.AdmissionChecksStrategy = roundTripped.Spec.AdmissionChecksStrategy
+				if diff := cmp.Diff(expected, roundTripped); diff != "" {
+					t.Errorf("round-trip conversion produced diff (-expected +roundtripped):\n%s", diff)
+				}
