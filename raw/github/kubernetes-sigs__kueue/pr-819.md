+# rayjob-code-sample.yaml

+kind: ConfigMap

+  name: ray-job-code-sample
+data:
+  sample_code.py: |
+    import ray
+    import os
+    import requests
+
+    ray.init()
+
+    @ray.remote
+    class Counter:
+        def __init__(self):

+            self.name = os.getenv("counter_name")
+            self.counter = 0
+

+            self.counter += 1
+
+        def get_counter(self):
