### Comment by [@mszadkow](https://github.com/mszadkow) — 2026-03-02T20:32:14Z

> > I introduced a side cache for the last seen cohort because integration tests showed that refreshing old cohort metrics is more reliable in Reconcile. When the refresh was done in Update, metrics for the old cohort could be re-recorded with stale data due to the cache not being up to date.
> 
> Hm, not sure I understand. My proposal in one of the comments was to update the metric in the Reconcile, with the exception of detecting the `cohortName` change - in that case just Clear the metric for old cohort in the Update. The value for the new cohortName would have been added in the Reconcile.
> 
> I'm not trilled by the lastSeenCohort, this introduces extra complexity, so exploring why/if needed.

I added the side cache for the last seen cohort because just clearing the old cohort’s metric in Update wasn’t reliable. The problem is, if the cache isn’t updated yet or if another controller reconciles at the wrong time, the old metric can get set again with stale data. That’s what was causing the inconsistent results in the integration tests.

With the side cache, we can detect cohort changes in Reconcile, after the cache is up to date. This way, we only refresh metrics when it’s safe and the state is consistent, which avoids those race conditions and keeps the metrics accurate. It does add a bit of complexity, but it makes the whole process much more robust.

I will make additional tests in the cluster to show exact behaviour in the wild.

+	flavorResource resources.FlavorResource
+	qty            int64
+}
+
+func (c *Cache) RecordCohortMetrics(log logr.Logger, cohortName kueue.CohortReference) {
+	if cohortName == "" {
+		log.V(4).Info("Cohort name is empty, skipping metrics recording")
+		return
+	}
+	log = c.withCohortLogger(log, cohortName)
+	log.V(4).Info("Recording metrics for cohort")
+
+	points := c.collectCohortMetricPoints(cohortName)
+	for _, p := range points {
+		c.applyCohortMetricPoint(p)
+	}
+}
+
+func (c *Cache) ClearCohortMetrics(log logr.Logger, cohortName kueue.CohortReference) {
+	if cohortName == "" {
+		log.V(4).Info("Cohort name is empty, skipping clearing metrics")
+		return
+	}
+
+	log = c.withCohortLogger(log, cohortName)
+	log.V(4).Info("Clearing metrics for cohort")
+
+	points := c.collectCohortClearPoints(cohortName)
+	for _, p := range points {
+		c.applyCohortMetricPoint(p)
+	}
+}
+
+func (c *Cache) collectCohortMetricPoints(cohortName kueue.CohortReference) []cohortMetricPoint {
+	c.RLock()
+	defer c.RUnlock()
+
+	ch := c.hm.CohortByName(cohortName)
+	if ch == nil || hierarchy.HasCycle(ch) {
+		return nil
+	}
+
+	var points []cohortMetricPoint
+	for ancestor := range ch.PathSelfToRoot() {
+		for flr, qty := range ancestor.resourceNode.SubtreeQuota {
+			points = append(points, cohortMetricPoint{cohortName: ancestor.Name, flavorResource: flr, qty: qty})
+		}
+	}
+	return points
+}
+
+func (c *Cache) collectCohortClearPoints(cohortName kueue.CohortReference) []cohortMetricPoint {
+	c.RLock()
+	defer c.RUnlock()
+
+	cohort := c.hm.CohortByName(cohortName)
+	// If the cohort is not found, treat it as an implicit cohort removal
+	// and return a point that clears all metrics for that cohort label.
+	if cohort == nil {
+		return []cohortMetricPoint{{cohortName: cohortName}}
+	}
+
+	removedQuota := cohort.resourceNode.SubtreeQuota
+	var points []cohortMetricPoint
+	for ancestor := range cohort.PathSelfToRoot() {
+		for flr, qty := range removedQuota {
+			newQty := max(ancestor.resourceNode.SubtreeQuota[flr]-qty, 0)
+			points = append(points, cohortMetricPoint{
+				cohortName:     ancestor.Name,
+				flavorResource: flr,
+				qty:            newQty,
+			})
+		}
+	}
+	return points
+}
