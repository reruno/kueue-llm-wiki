  I would lean towards the struct-based handler pattern which is more idiomatic for controller-runtime.
  ```go
  type workloadHandler struct{}
  
  func (h *workloadHandler) Create(_ context.Context, _ event.CreateEvent, _ workqueue.TypedRateLimitingInterface[reconcile.Request]) {
  }
  
  func (h *workloadHandler) Update(_ context.Context, _ event.UpdateEvent, _ workqueue.TypedRateLimitingInterface[reconcile.Request]) {
  }
  
  func (h *workloadHandler) Delete(_ context.Context, e event.DeleteEvent, q workqueue.TypedRateLimitingInterface[reconcile.Request]) {
  	wl, ok := e.Object.(*kueue.Workload)
  	if !ok {
  		return
  	}
  	ownerName, ok := wl.Annotations[constants.JobOwnerNameAnnotation]
  	if !ok {
  		return
  	}
  	ownerGVK, ok := wl.Annotations[constants.JobOwnerGVKAnnotation]
  	if !ok || ownerGVK != gvk.String() {
  		return
  	}
  	q.Add(reconcile.Request{
  		NamespacedName: types.NamespacedName{
  			Name:      ownerName,
  			Namespace: wl.Namespace,
  		},
  	})
  }
  
  func (h *workloadHandler) Generic(_ context.Context, _ event.GenericEvent, _ workqueue.TypedRateLimitingInterface[reconcile.Request]) {
  }
  ```
